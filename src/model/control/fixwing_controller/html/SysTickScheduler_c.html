8unsigned long schdl_counter;



9unsigned long schdl_counter_max;



10



11extern volatile uint32_t taskToSchedule;



12



13static void SysTick_Isr(void)



14{



15    __DSB();



16    __ISB();



17	



18	if (schdl_counter >= schdl_counter_max)



19	{



20		schdl_counter = 0;



21	}



22	



23    if (0 == schdl_counter++)	/* Compare with zero before increment */



24	{



25		/* The rt_OneStep() function expects interrupts to be disabled. 



26           Hence disable interrupt before rt_OneStep(). */



27        __disable_irq();



28        rt_OneStep();



29        /* The rt_OneStep() function disables interrupt. 



30           Hence enable interrupt immediately after rt_OneStep(). */



31        __enable_irq();



32	}



33}



34



35#if (__CORTEX_M >= 3)



36    void SysTick_Handler(){



37    SCB->ICSR |= SCB_ICSR_PENDSVSET_Msk;



38}



39#else



40MW_RENTRANT_ISR_FUNCTION_BEGIN(SysTick_Handler, -1)



41    MW_CALL_ISR_FUNCTION(SysTick_Isr);



42MW_RENTRANT_ISR_FUNCTION_END()



43#endif

45void ARMCM_SysTick_Config(float modelBaseRate)



46{



47#if !defined(MW_DONT_USE_SYSTICK)    



48#if defined(MW_SCHEDULER_INTERRUPT_SOURCE) && (MW_SCHEDULER_INTERRUPT_SOURCE == 0)



49	/* SysTick Scheduler is selected */



50	unsigned long tick_counter;



51	



52	tick_counter = (unsigned long)(SystemCoreClock * modelBaseRate);



53	



54	if (tick_counter > SysTick_LOAD_RELOAD_Msk)



55	{



56		schdl_counter_max = (unsigned long)((float)tick_counter / (float)SysTick_LOAD_RELOAD_Msk) + 1;



57		SysTick_Config(tick_counter/schdl_counter_max);		



58	}



59	else



60	{



61		schdl_counter_max = 0;



62		SysTick_Config(tick_counter);



63	}



64	



65	schdl_counter = 0;



66	



67	/* Set the SysTick priority */



68	#ifdef MW_SCHEDULER_PRIORITY



69		NVIC_SetPriority((IRQn_Type)(-1), MW_SCHEDULER_PRIORITY);



70	#endif



71    #if (__CORTEX_M >= 3)



72        taskToSchedule = (uint32_t) SysTick_Isr;



73    #endif



74#endif



75#endif



76    NVIC_SetPriority((IRQn_Type)(-5), MW_MAX_PRIORITY_VALUE);



77    NVIC_SetPriority((IRQn_Type)(-2), MW_MAX_PRIORITY_VALUE);



78}

80void ARMCM_SysTick_Stop(void)



81{



82#if !defined(MW_DONT_USE_SYSTICK)    



83#if defined(MW_SCHEDULER_INTERRUPT_SOURCE) && (MW_SCHEDULER_INTERRUPT_SOURCE == 0)



84	/* SysTick Scheduler is Stopped */



85	SysTick->CTRL &= ~SysTick_CTRL_ENABLE_Msk;



86#endif



87#endif



88}
