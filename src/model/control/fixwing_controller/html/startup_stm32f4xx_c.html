136/* Startup routine */



137void Default_ResetHandler(void)



138{



139#if defined(__GNUC__) && !defined(MW_ARM_CLANG)



140  /* Initialize data and bss */



141  unsigned long *pulSrc, *pulDest;



142



143  /* Copy the data segment initializers from flash to SRAM */



144  pulSrc = &_sidata;



145



146  for(pulDest = &_sdata; pulDest < &_edata; ) {



147      *(pulDest++) = *(pulSrc++);



148  }



149  



150  



151  /* Clear .bss */



152  __asm("  ldr     r0, =_sbss\n"



153        "  ldr     r1, =_ebss\n"



154        "  mov     r2, #0\n"



155        "  .thumb_func\n"



156        "zero_loop:\n"



157        "    cmp     r0, r1\n"



158        "    it      lt\n"



159        "    strlt   r2, [r0], #4\n"



160        "    blt     zero_loop");



161#endif



162	



163#ifndef __NO_SYSTEM_INIT



164  /* Call CMSIS system initialization code */



165  SystemInit();



166#endif



167  



168  /* Branch to main */



169#ifndef __START



170#define __START main



171#endif



172



173  __START();



174}

181	{



182	}
